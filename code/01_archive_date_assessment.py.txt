# list all the archive_on dates of HES APC
hes_apc_arch = spark.table(f'''{dbc}.hes_apc_all_years_archive''')
vector_dates = list(hes_apc_arch.\
    select(f.col("archived_on")).distinct().collect())
list_dates = [vec["archived_on"] for vec in vector_dates]
