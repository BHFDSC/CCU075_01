# MAGIC - Clean HES OP

# MAGIC - Find the earliest date where MAINSPEF or TRETSPEF is in the values specified

# COMMAND ----------

display(df_hes_op.limit(5))

# COMMAND ----------

# Make a clean HES OP table in long format for phenotype extraction

df_hes_op_sel = df_hes_op.filter(f.col("PERSON_ID_DEID").isNotNull()).\
    filter(f.col("APPTDATE").isNotNull()).\
        select(["PERSON_ID_DEID", "APPTDATE", "MAINSPEF", "TRETSPEF",'PROCODE3', 'PROCODE5']).\
  withColumnRenamed("PROCODE3", "OP_spec_PROCODE3").withColumnRenamed("PROCODE5", "OP_spec_PROCODE5")

# COMMAND ----------

df_temp_qc = df_tak.select(["PERSON_ID_DEID", "dob_apc_op"])
df_join_op_qc = df_temp_qc.join(df_hes_op_sel, on= "PERSON_ID_DEID", how="left") 

# COMMAND ----------

# any apptdate < birthdate?
print(df_join_op_qc.filter(f.col("APPTDATE")< f.col("dob_apc_op")).count())

# COMMAND ----------

# filter 
df_join_op_qc = df_join_op_qc.filter(f.col("APPTDATE")>=f.col("dob_apc_op"))

# COMMAND ----------

# Keep only selected MAINSTEP/TRETSPEF
spec_list = ["410", "361", "107"]
df_specialty = df_join_op_qc.filter(\
    (f.col("MAINSPEF").isin(spec_list)) | \
    (f.col("TRETSPEF").isin(spec_list)))

# COMMAND ----------

# Keep the earliest attendance 
w_spec = Window.partitionBy("PERSON_ID_DEID").orderBy(f.col("APPTDATE").asc())
df_specialty = df_specialty.withColumn("spec_rank", f.row_number().over(w_spec))

# COMMAND ----------

display(df_specialty.limit(10))

# COMMAND ----------

df_specialty = df_specialty.\
    filter(f.col("spec_rank")==1).\
        withColumnRenamed("APPTDATE", "OP_spec_date").\
            select(["PERSON_ID_DEID", "OP_spec_date", "MAINSPEF", "TRETSPEF", "OP_spec_PROCODE3", "OP_spec_PROCODE5"])

# COMMAND ----------

df_specialty = df_specialty.withColumn("flag_spec_hes_op", f.lit(1))

# COMMAND ----------

df_out = df_tak.select("PERSON_ID_DEID", "dob_apc_op").join(df_specialty, on=["PERSON_ID_DEID"],how="left")
# age at OP specialityvisit event date 
df_out = df_out.withColumn("OP_age_spec", f.floor(
    f.datediff(
        f.col("OP_spec_date"), f.col("dob_apc_op")
    )/365.25
))
df_out = df_out.drop("dob_apc_op")

df_out = df_out.fillna({"flag_spec_hes_op":0})        
display(df_out.limit(5))                  

# COMMAND ----------

print(df_tak.select("PERSON_ID_DEID").distinct().count())
print(df_out.select("PERSON_ID_DEID").distinct().count())

# COMMAND ----------

display(df_out.groupBy(["flag_spec_hes_op"]).agg(f.count("*")))

# COMMAND ----------
