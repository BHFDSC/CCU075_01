# MAGIC %md
# MAGIC # Extract ICD10 from HES APC

# COMMAND ----------

columns = ['name', 'terminology', 'CODE', 'term']
pheno_ta_hes = [('TA_HES','ICD10','I351','Aortic (valve) insufficiency'),
('TA_HES','ICD10','I652','Occlusion and stenosis of carotid artery'),
('TA_HES','ICD10','I712','Thoracic aortic aneurysm, without mention of rupture'),
('TA_HES','ICD10','I771','Stricture of artery'),
('TA_HES','ICD10','I776','Arteritis, unspecified'),
('TA_HES','ICD10','M316','Other giant cell arteritis'),
('TA_HES','ICD10','Z512','Other chemotherapy'),
('TA_HES','ICD10','Z958','Presence of other cardiac and vascular implants and grafts')]
df_codelist = spark.createDataFrame(pheno_ta_hes, columns)
display(df_codelist)

# COMMAND ----------

# Make a clean HES APC table in long format for phenotype extraction

df_hes_apc_sel = df_hes_apc.filter(f.col("PERSON_ID_DEID").isNotNull()).\
    filter(f.col("EPISTART").isNotNull()).\
        select(["PERSON_ID_DEID", "EPISTART", "EPIKEY", "PROCODE3","PROCODE5",  
                 f.split("DIAG_4_CONCAT", ",").alias("DIAG_4_ARRAY")])

# COMMAND ----------

df_long_4 = df_hes_apc_sel.select("PERSON_ID_DEID", "EPISTART",  "PROCODE3","PROCODE5",f.posexplode("DIAG_4_ARRAY").alias("DIAG_POSITION", "CODE")).\
    withColumn("DIAG_DIGITS", f.lit(4)).withColumn("DIAG_POSITION", f.expr("DIAG_POSITION + 1")).\
    withColumn("CODE", f.regexp_replace("CODE", r"[.,\-\s]", "")).\
    filter((f.col("CODE").isNotNull()) & (f.col("CODE") != ""))
display(df_long_4.limit(200))

# COMMAND ----------

# Join with the codelist

df_join_hes = df_long_4.join(df_codelist, on= ["CODE"], how= "inner")

# COMMAND ----------

# MAGIC %md
# MAGIC  Strategy: 
# MAGIC  - Load the dob_curated (DOB from HES APC)
# MAGIC  - Join with  HES APC
# MAGIC  - Drop records with Epistart < dob_curated (if any)
# MAGIC  - After QC, keep the minimum date per person
# MAGIC  - Also keep PROCODE3 and PROCODE5 
# MAGIC  - add a flag
# MAGIC  - Create the age at HES APC ICD10 event date

# COMMAND ----------

# DOB  (for quality control of ICD10 event dates)
df_dob_hes_tak = df_tak.filter(f.col("flag_hes_apc_tak")==1).select(["PERSON_ID_DEID", "dob_apc_op"])

# COMMAND ----------

df_join_hes_qc = df_dob_hes_tak.join(df_join_hes, on= "PERSON_ID_DEID", how="left") 

# COMMAND ----------

# any epistart < birthdate?
print(df_join_hes_qc.filter(f.col("EPISTART")< f.col("dob_apc_op")).count())

# COMMAND ----------

# filter these out
df_join_hes_qc = df_join_hes_qc.filter(f.col("EPISTART")>=f.col("dob_apc_op"))

# COMMAND ----------

# Keep the minimum EPISTART 
w_1 = Window.partitionBy("PERSON_ID_DEID").orderBy(f.col("EPISTART").asc())
df_join_hes_qc = df_join_hes_qc.withColumn("epi_order", f.row_number().over(w_1))
df_first_event = df_join_hes_qc.filter(f.col("epi_order")==1)

# COMMAND ----------

display(df_first_event.limit(10)) 

# COMMAND ----------

# Select columns, rename EPISTART, and join back with df_cohort
df_first_event_sel = df_first_event.select(["PERSON_ID_DEID", "EPISTART", "CODE", "PROCODE3", "PROCODE5"]).\
    withColumnRenamed("EPISTART", "APC_ICD10_CODE_date").\
        withColumnRenamed("CODE", "APC_ICD10_code").\
            withColumnRenamed("PROCODE3", "APC_ICD10_PROCODE3").\
                withColumnRenamed("PROCODE5", "APC_ICD10_PROCODE5")


df_first_event_sel = df_first_event_sel.withColumn("flag_icd10_hes_apc", f.lit(1))

# COMMAND ----------

df_out  = df_tak.select(["PERSON_ID_DEID", "dob_apc_op"]).join(df_first_event_sel,
                             on= ["PERSON_ID_DEID"], how="left")
df_out = df_out.withColumn("APC_age_ICD10", f.floor(
    f.datediff(
        f.col("APC_ICD10_CODE_date"), f.col("dob_apc_op")
    )/365.25
))

df_out = df_out.drop("dob_apc_op")
df_out = df_out.fillna({"flag_icd10_hes_apc":0})                            
display(df_out.limit(5))
