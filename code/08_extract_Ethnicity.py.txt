df_mix_apc = df_mix.filter(f.col("flag_hes_apc_tak")==1)

# MAGIC %md
# MAGIC # Ethnicity from HES AP

# COMMAND ----------

# Extract IDs to join with HES APC
ids_hes_apc = df_mix_apc.select("PERSON_ID_DEID")

# COMMAND ----------

# Left join with hes for ethnicity
df_eth_apc_1 = ids_hes_apc.join(df_hes_apc.select(["PERSON_ID_DEID", "ETHNOS", "EPISTART"]), on=["PERSON_ID_DEID"], how="left" )

# COMMAND ----------

# Ethnic codes
display(df_eth_apc_1.groupby("ETHNOS").agg(f.count("*")))

# COMMAND ----------

# MAGIC %md
# MAGIC Ethnic codes in data dictionary

# MAGIC - E:White and Black African (Mixed) (Mixed)

# COMMAND ----------

# Test of duplicated epistart
w_test = Window.partitionBy(["PERSON_ID_DEID", "EPISTART"]).orderBy(f.col("EPISTART"))
df_tst = df_eth_apc_1.withColumn("rank", f.row_number().over(w_test))
display(df_tst.filter(f.col('rank')>1))

# COMMAND ----------

# MAGIC %md
# MAGIC # Strategy for ethnicity
# MAGIC
# MAGIC - Keep unique combinations of epistart and ethnos
# MAGIC   - e.g., if ethnicity A is recorded twice on 2023-12-21, keep only one
# MAGIC - Count each ethnicity per person
# MAGIC - Sort by the count (descending) and epistart (ascnding)
# MAGIC - Select the first ethnicity 

# COMMAND ----------

# Keep unique combination of Ethnicity and epistart 
# The orderby does not add anything to the logic of the Window. It is just needed by the row_number()
w_dup = Window.partitionBy(["PERSON_ID_DEID", "ETHNOS", "EPISTART"]).orderBy(f.col("EPISTART"))
df_eth_apc_2 = df_eth_apc_1.withColumn("eth_order", f.row_number().over(w_dup))
df_eth_apc_2 = df_eth_apc_2.filter(f.col("eth_order")==1).drop("eth_order")

# COMMAND ----------

# Define the window funciton for counting different ethnicities per person
w_eth_2= Window.partitionBy(["PERSON_ID_DEID", "ETHNOS"])
df_eth_apc_3 = df_eth_apc_2.withColumn("eth_count", f.count("ETHNOS").over(w_eth_2))

# COMMAND ----------

# Keep one row per person+ethnos with most recent EPISTART
w_eth_3= Window.partitionBy(["PERSON_ID_DEID", "ETHNOS"]).orderBy(f.col("EPISTART").desc_nulls_last())
df_eth_apc_4 = df_eth_apc_3.withColumn("eth_rank", f.row_number().over(w_eth_3))
df_eth_apc_4 = df_eth_apc_4.filter(f.col("eth_rank")==1).drop("eth_rank")

# COMMAND ----------

# Order by EPISTART (descending)+count (descending)  and keep the first row
w_eth_4 = Window.partitionBy(["PERSON_ID_DEID"]).orderBy(f.col("eth_count").desc(), f.col("EPISTART").desc_nulls_last())
df_eth_apc_5 = df_eth_apc_4.withColumn("final_rank", f.row_number().over(w_eth_4))

# COMMAND ----------

# final table 
df_eth_apc_final = df_eth_apc_5.filter(f.col("final_rank")==1).drop("eth_count").drop("final_rank").drop("EPISTART")

# COMMAND ----------

df_eth_apc_final.cache()

# COMMAND ----------

print(df_eth_apc_final.select("PERSON_ID_DEID").count())
print(df_eth_apc_final.select("PERSON_ID_DEID").distinct().count())

# COMMAND ----------

# Any null ethnicities?
df_eth_apc_final.filter(f.col("ETHNOS").isNull()).count()

# COMMAND ----------

# Ethnic codes
display(df_eth_apc_final.groupby("ETHNOS").agg(f.count("*")))

# COMMAND ----------

# COMMAND ----------

# Extract IDs to join with HES OP
ids_hes_op = df_mix_op.select("PERSON_ID_DEID")

# COMMAND ----------

# Left join with hes for ethnicity
df_eth_op_1 = ids_hes_op.join(df_hes_op.select(["PERSON_ID_DEID", "ETHNOS", "APPTDATE"]), on=["PERSON_ID_DEID"], how="left" )

# COMMAND ----------

# Ethnic codes
display(df_eth_op_1.groupby("ETHNOS").agg(f.count("*")))

# COMMAND ----------

# Test of duplicated epistart
w_test_op = Window.partitionBy(["PERSON_ID_DEID", "APPTDATE"]).orderBy(f.col("APPTDATE"))
df_tst_op = df_eth_op_1.withColumn("rank", f.row_number().over(w_test_op))
display(df_tst_op.filter(f.col('rank')>1))

# COMMAND ----------

# Keep unique combination of Ethnicity and APPTDATE 
# The orderby does not add anything to the logic of the Window. It is just needed by the row_number()
w_dup_op = Window.partitionBy(["PERSON_ID_DEID", "ETHNOS", "APPTDATE"]).orderBy(f.col("APPTDATE"))
df_eth_op_2 = df_eth_op_1.withColumn("eth_order", f.row_number().over(w_dup_op))
df_eth_op_2 = df_eth_op_2.filter(f.col("eth_order")==1).drop("eth_order")

# COMMAND ----------

# Define the window funciton for counting different ethnicities per person
w_eth_2_op= Window.partitionBy(["PERSON_ID_DEID", "ETHNOS"])
df_eth_op_3 = df_eth_op_2.withColumn("eth_count", f.count("ETHNOS").over(w_eth_2_op))

# COMMAND ----------

# Keep one row per person+ethnos with most recent APPTDATE
w_eth_3_op= Window.partitionBy(["PERSON_ID_DEID", "APPTDATE"]).orderBy(f.col("APPTDATE").desc_nulls_last())
df_eth_op_4 = df_eth_op_3.withColumn("eth_rank", f.row_number().over(w_eth_3_op))
df_eth_op_4 = df_eth_op_4.filter(f.col("eth_rank")==1).drop("eth_rank")

# COMMAND ----------

# Order by APPTDATE (descending)+count (descending)  and keep the first row
w_eth_4_op = Window.partitionBy(["PERSON_ID_DEID"]).orderBy(f.col("eth_count").desc(), f.col("APPTDATE").desc_nulls_last())
df_eth_op_5 = df_eth_op_4.withColumn("final_rank", f.row_number().over(w_eth_4_op))

# COMMAND ----------

# final table 
df_eth_op_final = df_eth_op_5.filter(f.col("final_rank")==1).drop("eth_count").drop("final_rank").drop("APPTDATE")

# COMMAND ----------

df_eth_op_final.cache()

# COMMAND ----------

print(df_eth_op_final.select("PERSON_ID_DEID").count())
print(df_eth_op_final.select("PERSON_ID_DEID").distinct().count())

# COMMAND ----------

# Any null ethnicities?
df_eth_op_final.filter(f.col("ETHNOS").isNull()).count()

# COMMAND ----------

# Ethnic codes
display(df_eth_op_final.groupby("ETHNOS").agg(f.count("*")))

# MAGIC %md
# MAGIC
# MAGIC # Union of HES APC and HES OP ethnicities and join with the cohort

# COMMAND ----------

df_eth_apc_final = df_eth_apc_final.withColumn("ethnicity_source", f.lit("HES APC"))
df_eth_op_final = df_eth_op_final.withColumn("ethnicity_source", f.lit("HES OP"))

df_eth_union = df_eth_apc_final.union(df_eth_op_final)

# COMMAND ----------

display(df_eth_union.groupBy("ethnicity_source").agg(f.count("*")))
